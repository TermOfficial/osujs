/**
 * Initializes osujs, setting up the stage with the specified canvas dimensions and fps.
 *
 * @method initialize
 * @param canvas {String} The id of the canvas element that osujs will draw to.
 * @param fps {Number} Optional parameter to specify the fps for osujs to run at (defaults to 120).
 **/

// namespacing
this.osujs = this.osujs || {};

(function () {

  // all internal calculations are processed at a resolution of 1024x768
  var internalResolution = { width : 1024, height : 768 };
  var defaultFPS = 120;
  
  function initialize(canvas, fps) {
    // initialize createjs stuff
    createjs.Ticker.setFPS(fps || defaultFPS);
    createjs.Ticker.useRAF = 1;
    createjs.Ticker.init();

    // globally accesible stage
    osujs.stage = new createjs.Stage(canvas);
    osujs.stage.scaleX = internalResolution.width/osujs.stage.canvas.width;
    osujs.stage.scaleY = internalResolution.height/osujs.stage.canvas.height;
    osujs.stage.handleEvent = function (e) {
      switch (e) {
        case "tick":
          osujs.stage.update();
          break;
      };
    };
    createjs.Ticker.addEventListener("tick", osujs.stage);
  };

  osujs.initialize = initialize;

}());
